import pandas as pd
'''
Input: filename (str) - the name of the input CSV file.
Output: actors_movies (dict) - a dictionary where the keys are actor names and the values are lists of movies they have acted in.
'''
def get_actor_movie_list(filename):
    df = pd.read_csv(filename)
    actors_movies = {}
    for i in range(len(df)):
        actors = df.loc[i, 'actors']
        movie = df.loc[i, 'movie_name']
        for actor in actors:
            if actor not in actors_movies:
                actors_movies[actor] = [movie]
            else:
                actors_movies[actor].append(movie)
    return actors_movies
